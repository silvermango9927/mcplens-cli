# agentify V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `agentify`, a CLI that compiles an OpenAPI spec + sample payloads into a lean TypeScript MCP server, using Claude once at build time and zero LLM calls at runtime.

**Architecture:** Two-phase pipeline. `compile` (the only LLM step): spec-loader → sampler → analyzer → `agentify.manifest.json`. `build` (deterministic): manifest → codegen → generated MCP server project → verifier (tsc + stdio smoke test, LLM repair loop on failure). The manifest is the single contract between the halves.

**Tech Stack:** Node 18+, TypeScript (ESM, NodeNext), commander, @apidevtools/swagger-parser, @anthropic-ai/sdk, @modelcontextprotocol/sdk (used both in generated output and by the verifier's smoke-test client), zod, vitest, tsx.

**Spec:** `docs/superpowers/specs/2026-06-12-agentify-v1-design.md`

**Conventions locked in here:**
- Sample file shape: `{ "endpoint": "GET /issues/{id}", "response": { … } }`.
- Tool requests: exactly one with `key: "main"`. Secondary requests run iff the
  auto-generated `include` input array contains their key. Tool-level `query`/`body`
  params apply to the main request only; all requests get path interpolation from inputs.
- `responseMap.from` paths: dot segments; a segment ending `[]` maps over an array.
  Transforms apply element-wise over arrays (except `count`).
- LLM outputs only `{ tools, hiddenEndpoints }` (PartialAnalysis). The `api` block
  (name/baseUrl/auth/env-var names) is assembled in code, never by the LLM.

```ts
import { z } from 'zod'

export const TRANSFORM_NAMES = ['stripHtml', 'adfToPlainText', 'toString', 'count', 'firstLine'] as const
export type TransformName = (typeof TRANSFORM_NAMES)[number]

export const ResponseMapEntrySchema = z.object({
  from: z.string().min(1),
  to: z.string().min(1),
  source: z.string().min(1).default('main'),
  transform: z.enum(TRANSFORM_NAMES).optional(),
  reason: z.string().default('')
})

export const ToolParamSchema = z.object({
  name: z.string().min(1),
  in: z.enum(['path', 'query', 'body']),
  type: z.enum(['string', 'number', 'boolean', 'string[]']),
  required: z.boolean().default(false),
  description: z.string().default('')
})

export const ToolRequestSchema = z.object({
  key: z.string().min(1),
  method: z.enum(['GET', 'POST', 'PUT', 'PATCH', 'DELETE']),
  path: z.string().min(1)
})

export const ToolSchema = z
  .object({
    name: z.string().regex(/^[a-z][a-z0-9_]*$/),
    description: z.string().min(1),
    params: z.array(ToolParamSchema).default([]),
    requests: z.array(ToolRequestSchema).min(1),
    responseMap: z.array(ResponseMapEntrySchema).min(1)
  })
  .superRefine((tool, ctx) => {
    const issue = (message: string) => ctx.addIssue({ code: z.ZodIssueCode.custom, message })
    const keys = tool.requests.map((r) => r.key)
    if (tool.requests.filter((r) => r.key === 'main').length !== 1) issue('exactly one request must have key "main"')
    if (new Set(keys).size !== keys.length) issue('request keys must be unique')
    if (tool.requests.length > 1 && tool.params.some((p) => p.name === 'include'))
      issue('"include" is a reserved param name on multi-request tools')
    for (const e of tool.responseMap)
      if (!keys.includes(e.source)) issue(`responseMap source "${e.source}" has no matching request`)
  })

export const AuthSchema = z.discriminatedUnion('type', [
  z.object({ type: z.literal('bearer'), envVar: z.string().min(1) }),
  z.object({ type: z.literal('header'), header: z.string().min(1), envVar: z.string().min(1) }),
  z.object({ type: z.literal('basic'), userEnvVar: z.string().min(1), passEnvVar: z.string().min(1) }),
  z.object({ type: z.literal('none') })
])

export const HiddenEndpointSchema = z.object({ endpoint: z.string().min(1), reason: z.string().default('') })

export const ManifestSchema = z.object({
  agentifyVersion: z.literal(1),
  api: z.object({ name: z.string().min(1), baseUrl: z.string().url(), auth: AuthSchema }),
  tools: z.array(ToolSchema).min(1),
  hiddenEndpoints: z.array(HiddenEndpointSchema).default([])
})

// What the LLM is allowed to produce — everything else is assembled in code.
export const PartialAnalysisSchema = z.object({
  tools: z.array(ToolSchema).min(1),
  hiddenEndpoints: z.array(HiddenEndpointSchema).default([])
})

export type Manifest = z.infer<typeof ManifestSchema>
export type ManifestTool = z.infer<typeof ToolSchema>
export type ManifestAuth = z.infer<typeof AuthSchema>
export type ResponseMapEntry = z.infer<typeof ResponseMapEntrySchema>
export type ToolParam = z.infer<typeof ToolParamSchema>
export type PartialAnalysis = z.infer<typeof PartialAnalysisSchema>
```

`src/mapping/runtime.ts`:
```ts
// Lean-response mapping runtime. Dependency-free and self-contained on purpose:
// agentify copies this file verbatim into every generated MCP server (src/lib/mapping.ts).

export type TransformName = 'stripHtml' | 'adfToPlainText' | 'toString' | 'count' | 'firstLine'

export interface MapEntry {
  from: string
  to: string
  source?: string
  transform?: TransformName
  reason?: string
}

export function getPath(value: unknown, path: string): unknown {
  if (!path) return value
  const [head, ...rest] = path.split('.')
  const restPath = rest.join('.')
  if (head.endsWith('[]')) {
    const arr = (value as Record<string, unknown> | null | undefined)?.[head.slice(0, -2)]
    if (!Array.isArray(arr)) return undefined
    return arr.map((item) => getPath(item, restPath))
  }
  const next = (value as Record<string, unknown> | null | undefined)?.[head]
  if (rest.length === 0) return next
  return getPath(next, restPath)
}

function adfToPlainText(node: unknown): string {
  const out: string[] = []
  const walk = (n: unknown): void => {
    if (n == null || typeof n !== 'object') return
    const rec = n as { text?: unknown; content?: unknown; type?: unknown }
    if (typeof rec.text === 'string') out.push(rec.text)
    if (Array.isArray(rec.content)) {
      rec.content.forEach(walk)
      if (rec.type === 'paragraph') out.push('\n')
    }
  }
  walk(node)
  return out.join('').trim()
}

export const transforms: Record<TransformName, (v: unknown) => unknown> = {
  stripHtml: (v) => (typeof v === 'string' ? v.replace(/<[^>]*>/g, '') : v),
  adfToPlainText: (v) => adfToPlainText(v),
  toString: (v) => (v == null ? v : String(v)),
  count: (v) => (Array.isArray(v) ? v.length : v == null ? 0 : 1),
  firstLine: (v) => (typeof v === 'string' ? v.split('\n')[0] : v)
}

export function applyResponseMap(
  payloads: Record<string, unknown>,
  entries: MapEntry[]
): Record<string, unknown> {
  const out: Record<string, unknown> = {}
  for (const e of entries) {
    const src = payloads[e.source ?? 'main']
    if (src === undefined) continue
    let v = getPath(src, e.from)
    if (e.transform) {
      const fn = transforms[e.transform]
      v = Array.isArray(v) && e.transform !== 'count' ? v.map(fn) : fn(v)
    }
    if (v !== undefined) out[e.to] = v
  }
  return out
}
```

```ts
export interface EndpointParam { name: string; in: string; required: boolean; type: string }
export interface EndpointSummary {
  method: string
  path: string
  summary: string
  params: EndpointParam[]
  responseFields: string[]
}
export interface LoadedSpec {
  title: string
  baseUrl?: string
  auth: ManifestAuth & { envVar?: string } // envVar filled in by assembleManifest
  endpoints: EndpointSummary[]
}

const METHODS = ['get', 'post', 'put', 'patch', 'delete'] as const
const MAX_FIELDS = 80
const MAX_DEPTH = 4

type AnySchema = Record<string, any>

function flattenSchema(schema: AnySchema | undefined, prefix: string, depth: number, out: string[]): void {
  if (!schema || depth > MAX_DEPTH || out.length >= MAX_FIELDS) return
  if (schema.type === 'array' || schema.items) {
    flattenSchema(schema.items, prefix ? `${prefix}[]` : '[]', depth, out)
    return
  }
  const props: Record<string, AnySchema> = schema.properties ?? {}
  for (const [key, sub] of Object.entries(props)) {
    if (out.length >= MAX_FIELDS) return
    const path = prefix ? `${prefix}.${key}` : key
    if (sub.properties || sub.items) flattenSchema(sub, path, depth + 1, out)
    else out.push(path)
  }
}

function inferAuth(api: AnySchema): ManifestAuth {
  const schemes: Record<string, AnySchema> = api.components?.securitySchemes ?? {}
  for (const s of Object.values(schemes)) {
    if (s.type === 'http' && s.scheme === 'bearer') return { type: 'bearer', envVar: '' }
    if (s.type === 'http' && s.scheme === 'basic') return { type: 'basic', userEnvVar: '', passEnvVar: '' }
    if (s.type === 'apiKey' && s.in === 'header') return { type: 'header', header: s.name, envVar: '' }
  }
  return { type: 'none' }
}

export async function loadSpec(specPath: string): Promise<LoadedSpec> {
  let api: AnySchema
  try {
    api = (await SwaggerParser.dereference(specPath)) as AnySchema
  } catch (err) {
    throw new Error(`Failed to load OpenAPI spec at ${specPath}: ${(err as Error).message}`)
  }
  const endpoints: EndpointSummary[] = []
  for (const [path, item] of Object.entries<AnySchema>(api.paths ?? {})) {
    for (const method of METHODS) {
      const op = item?.[method]
      if (!op) continue
      const params: EndpointParam[] = [...(item.parameters ?? []), ...(op.parameters ?? [])].map((p: AnySchema) => ({
        name: p.name,
        in: p.in,
        required: Boolean(p.required),
        type: p.schema?.type ?? 'string'
      }))
      const bodySchema = op.requestBody?.content?.['application/json']?.schema
      if (bodySchema?.properties) {
        const required: string[] = bodySchema.required ?? []
        for (const [name, sub] of Object.entries<AnySchema>(bodySchema.properties))
          params.push({ name, in: 'body', required: required.includes(name), type: sub.type ?? 'string' })
      }
      const ok = op.responses?.['200'] ?? op.responses?.['201']
      const responseFields: string[] = []
      flattenSchema(ok?.content?.['application/json']?.schema, '', 0, responseFields)
      endpoints.push({ method: method.toUpperCase(), path, summary: op.summary ?? op.description ?? '', params, responseFields })
    }
  }
  return {
    title: api.info?.title ?? 'API',
    baseUrl: api.servers?.[0]?.url,
    auth: inferAuth(api),
    endpoints
  }
}
```

```ts
import type { EndpointSummary } from './loader.js'

export interface Batch { text: string; endpoints: EndpointSummary[] }

export function formatEndpoint(e: EndpointSummary): string {
  const params = e.params.map((p) => `${p.name}${p.required ? '*' : ''}(${p.type},${p.in})`).join(', ')
  const lines = [`${e.method} ${e.path}${e.summary ? ` — ${e.summary}` : ''}`]
  if (params) lines.push(`  params: ${params}`)
  if (e.responseFields.length) lines.push(`  response: ${e.responseFields.join(', ')}`)
  return lines.join('\n')
}

export function condenseEndpoints(endpoints: EndpointSummary[], maxChars = 300_000): Batch[] {
  const batches: Batch[] = []
  let cur: Batch = { text: '', endpoints: [] }
  for (const e of endpoints) {
    const block = formatEndpoint(e) + '\n\n'
    if (cur.text && cur.text.length + block.length > maxChars) {
      batches.push(cur)
      cur = { text: '', endpoints: [] }
    }
    cur.text += block
    cur.endpoints.push(e)
  }
  if (cur.endpoints.length) batches.push(cur)
  return batches
}
```

```ts
import { readFile, readdir } from 'node:fs/promises'
import { join } from 'node:path'
import type { ManifestAuth } from '../manifest/schema.js'
import type { LoadedSpec } from '../spec/loader.js'

export interface Sample { endpoint: string; response: unknown }

export async function loadSamplesDir(dir: string): Promise<Sample[]> {
  let files: string[]
  try {
    files = (await readdir(dir)).filter((f) => f.endsWith('.json'))
  } catch {
    return []
  }
  const samples: Sample[] = []
  for (const f of files) {
    const raw = JSON.parse(await readFile(join(dir, f), 'utf8')) as Partial<Sample>
    if (typeof raw.endpoint === 'string' && 'response' in raw) samples.push(raw as Sample)
    else console.warn(`agentify: skipping ${f} — expected { "endpoint": "...", "response": ... }`)
  }
  return samples
}

export function buildAuthHeaders(auth: ManifestAuth, env: Record<string, string | undefined>): Record<string, string> {
  const need = (name: string): string => {
    const v = env[name]
    if (!v) throw new Error(`Missing env var ${name} (required for ${auth.type} auth)`)
    return v
  }
  switch (auth.type) {
    case 'bearer': return { Authorization: `Bearer ${need(auth.envVar)}` }
    case 'header': return { [auth.header]: need(auth.envVar) }
    case 'basic': return { Authorization: `Basic ${Buffer.from(`${need(auth.userEnvVar)}:${need(auth.passEnvVar)}`).toString('base64')}` }
    case 'none': return {}
  }
}

export async function captureSamples(
  spec: LoadedSpec,
  baseUrl: string,
  headers: Record<string, string>,
  limit = 5,
  fetchImpl: typeof fetch = fetch
): Promise<Sample[]> {
  const candidates = spec.endpoints
    .filter((e) => e.method === 'GET' && !e.params.some((p) => p.in === 'path' && p.required))
    .slice(0, limit)
  const samples: Sample[] = []
  for (const e of candidates) {
    const url = baseUrl.replace(/\/$/, '') + e.path
    try {
      const res = await fetchImpl(url, { headers: { accept: 'application/json', ...headers } })
      if (!res.ok) {
        console.warn(`agentify: sample fetch ${e.method} ${e.path} -> ${res.status}, skipping`)
        continue
      }
      samples.push({ endpoint: `${e.method} ${e.path}`, response: await res.json() })
    } catch (err) {
      console.warn(`agentify: sample fetch ${e.method} ${e.path} failed: ${(err as Error).message}`)
    }
  }
  return samples
}
```

```ts
import type { Manifest } from '../manifest/schema.js'
import { applyResponseMap } from '../mapping/runtime.js'
import type { Sample } from '../sampler/sampler.js'

export function estimateTokens(s: string): number {
  return Math.ceil(s.length / 4)
}

export interface ToolSavings { tool: string; endpoint: string; rawTokens: number; leanTokens: number }
export interface SavingsReport { rows: ToolSavings[]; totalRaw: number; totalLean: number }

export function computeSavings(manifest: Manifest, samples: Sample[]): SavingsReport {
  const rows: ToolSavings[] = []
  for (const tool of manifest.tools) {
    const main = tool.requests.find((r) => r.key === 'main')!
    const endpoint = `${main.method} ${main.path}`
    const sample = samples.find((s) => s.endpoint === endpoint)
    if (!sample) continue
    const raw = estimateTokens(JSON.stringify(sample.response))
    const lean = estimateTokens(JSON.stringify(applyResponseMap({ main: sample.response }, tool.responseMap)))
    rows.push({ tool: tool.name, endpoint, rawTokens: raw, leanTokens: lean })
  }
  return {
    rows,
    totalRaw: rows.reduce((a, r) => a + r.rawTokens, 0),
    totalLean: rows.reduce((a, r) => a + r.leanTokens, 0)
  }
}

export function renderSavings(report: SavingsReport): string {
  if (!report.rows.length) return 'No samples matched tool endpoints — token-savings report unavailable.'
  const lines = ['Token savings (estimated, per call):']
  for (const r of report.rows) {
    const pct = r.rawTokens ? Math.round((1 - r.leanTokens / r.rawTokens) * 100) : 0
    lines.push(`  ${r.tool.padEnd(24)} ${String(r.rawTokens).padStart(7)} -> ${String(r.leanTokens).padStart(6)} tokens  (${pct}% smaller)  [${r.endpoint}]`)
  }
  const pct = report.totalRaw ? Math.round((1 - report.totalLean / report.totalRaw) * 100) : 0
  lines.push(`  TOTAL${' '.repeat(20)} ${String(report.totalRaw).padStart(7)} -> ${String(report.totalLean).padStart(6)} tokens  (${pct}% smaller)`)
  return lines.join('\n')
}
```

```ts
import { PartialAnalysisSchema, type PartialAnalysis } from '../manifest/schema.js'
import { condenseEndpoints } from '../spec/condense.js'
import type { LoadedSpec } from '../spec/loader.js'
import type { Sample } from '../sampler/sampler.js'
import { buildSystemPrompt, buildUserPrompt } from './prompt.js'

export interface CompletionDeps {
  complete: (system: string, user: string) => Promise<string>
}
export interface AnalyzeOptions { batchMaxChars?: number; maxAttempts?: number }

export function extractJson(text: string): unknown {
  const stripped = text.replace(/^\s*```(?:json)?\s*/i, '').replace(/\s*```\s*$/, '')
  return JSON.parse(stripped)
}

export async function analyze(
  spec: LoadedSpec,
  samples: Sample[],
  deps: CompletionDeps,
  opts: AnalyzeOptions = {}
): Promise<PartialAnalysis> {
  const { batchMaxChars = 300_000, maxAttempts = 3 } = opts
  const system = buildSystemPrompt()
  const batches = condenseEndpoints(spec.endpoints, batchMaxChars)
  const merged: PartialAnalysis = { tools: [], hiddenEndpoints: [] }
  const seenTools = new Set<string>()

  for (const batch of batches) {
    let user = buildUserPrompt(spec.title, batch.endpoints, samples)
    let lastError = ''
    let parsed: PartialAnalysis | undefined
    for (let attempt = 1; attempt <= maxAttempts; attempt++) {
      const raw = await deps.complete(system, user)
      try {
        parsed = PartialAnalysisSchema.parse(extractJson(raw))
        break
      } catch (err) {
        lastError = (err as Error).message
        user = `${buildUserPrompt(spec.title, batch.endpoints, samples)}\n\nYour previous output failed validation:\n${lastError.slice(0, 2000)}\nOutput ONLY corrected JSON.`
      }
    }
    if (!parsed) throw new Error(`Analyzer produced invalid output after ${maxAttempts} attempts: ${lastError.slice(0, 500)}`)
    for (const tool of parsed.tools) {
      if (seenTools.has(tool.name)) continue
      seenTools.add(tool.name)
      merged.tools.push(tool)
    }
    merged.hiddenEndpoints.push(...parsed.hiddenEndpoints)
  }
  return PartialAnalysisSchema.parse(merged)
}
```

```ts
export async function realExec(cmd: string, args: string[], cwd: string): Promise<ExecResult> {
  return new Promise((resolve) => {
    const child = spawn(cmd, args, { cwd, shell: process.platform === 'win32' })
    let output = ''

    child.stderr.on('data', (d) => (output += d))

  })

export async function realSmokeTest(dir: string): Promise<SmokeResult> {
  const { Client } = await import('@modelcontextprotocol/sdk/client/index.js')
  const { StdioClientTransport } = await import('@modelcontextprotocol/sdk/client/stdio.js')
  const transport = new StdioClientTransport({ command: 'node', args: ['dist/index.js'], cwd: dir })
  const client = new Client({ name: 'agentify-verifier', version: '0.1.0' })
  try {
    await client.connect(transport)
    const { tools } = await client.listTools()
    if (!tools.length) return { ok: false, error: 'server started but listed no tools' }
    return { ok: true, tools: tools.map((t) => t.name) }
  } catch (err) {
    return { ok: false, error: (err as Error).message }
  } finally {
    await client.close().catch(() => {})
  }
}

export async function verifyProject(
  dir: string,
  deps: VerifierDeps = { exec: realExec, smokeTest: realSmokeTest }
): Promise<VerifyResult> {
  const install = await deps.exec('npm', ['install', '--no-fund', '--no-audit'], dir)
  if (install.code !== 0) return { ok: false, stage: 'install', output: install.output }
  const tsc = await deps.exec('npx', ['tsc'], dir)
  if (tsc.code !== 0) return { ok: false, stage: 'tsc', output: tsc.output }
  const smoke = await deps.smokeTest(dir)
  if (!smoke.ok) return { ok: false, stage: 'smoke', output: smoke.error ?? 'unknown smoke failure' }
  return { ok: true, stage: 'done', output: `tools: ${(smoke.tools ?? []).join(', ')}` }
}
```

```ts
import { readFile, writeFile } from 'node:fs/promises'
import { join, normalize } from 'node:path'
import { z } from 'zod'
import type { CompletionDeps } from './analyzer.js'
import { extractJson } from './analyzer.js'

const RepairSchema = z.array(z.object({ path: z.string().min(1), content: z.string() }))

export function extractErrorFiles(tscOutput: string): string[] {
  const files = new Set<string>()
  for (const m of tscOutput.matchAll(/^([^\s(]+\.ts)\(\d+,\d+\): error/gm)) files.add(m[1])
  return [...files]
}

const SYSTEM = `You fix TypeScript compile errors in generated MCP server code. You receive failing files and tsc output. Return ONLY a JSON array: [{"path": "<same relative path>", "content": "<full corrected file>"}]. Change as little as possible. Never invent new files or paths.`

export async function repairProject(dir: string, tscOutput: string, deps: CompletionDeps): Promise<string[]> {
  const paths = extractErrorFiles(tscOutput)
  if (!paths.length) throw new Error(`tsc failed but no file paths found in output:\n${tscOutput.slice(0, 1000)}`)
  const fileBlocks = await Promise.all(
    paths.map(async (p) => `=== ${p} ===\n${await readFile(join(dir, p), 'utf8')}`)
  )
  const user = `tsc errors:\n${tscOutput.slice(0, 4000)}\n\nFailing files:\n${fileBlocks.join('\n\n')}`
  const fixes = RepairSchema.parse(extractJson(await deps.complete(SYSTEM, user)))
  const written: string[] = []
  for (const fix of fixes) {
    const normalized = normalize(fix.path)
    if (normalized.startsWith('..') || normalized.startsWith('/')) throw new Error(`Repair tried to write outside the project: ${fix.path}`)
    await writeFile(join(dir, normalized), fix.content)
    written.push(normalized)
  }
  return written
}
```

```ts
import { readFile } from 'node:fs/promises'
import { generateProject } from '../codegen/generate.js'
import { ManifestSchema } from '../manifest/schema.js'
import { verifyProject, type VerifyResult } from '../verifier/verify.js'

export interface BuildOptions {
  manifest: string
  out: string
  skipVerify?: boolean
  maxRepairs?: number
}
export interface BuildDeps {
  verify: (dir: string) => Promise<VerifyResult>
  repair: (dir: string, tscOutput: string) => Promise<string[]>
}

export async function runBuild(opts: BuildOptions, deps?: BuildDeps): Promise<VerifyResult> {
  const manifest = ManifestSchema.parse(JSON.parse(await readFile(opts.manifest, 'utf8')))
  await generateProject(manifest, opts.out)
  if (opts.skipVerify) return { ok: true, stage: 'done', output: 'generated (verification skipped)' }

  const verify = deps?.verify ?? verifyProject
  const maxRepairs = opts.maxRepairs ?? 2
  let result = await verify(opts.out)
  let repairs = 0
  while (!result.ok && result.stage === 'tsc' && repairs < maxRepairs && deps?.repair) {
    repairs += 1
    console.warn(`agentify: tsc failed, attempting LLM repair (${repairs}/${maxRepairs})…`)
    await deps.repair(opts.out, result.output)
    result = await verify(opts.out)
  }
  if (!result.ok) console.error(`agentify: verification failed at stage "${result.stage}". Artifacts kept in ${opts.out} for inspection.\n${result.output.slice(0, 2000)}`)
  return result
}
```
